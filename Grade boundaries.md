```
GET score 

if score > 80 THEN
    
    OUTPUT"You got an A-grade"


else if score > 70 THEN
    
    OUTPUT"You got a B-grade"


else if score > 60 THEN
    
    OUTPUT"You got a C-grade"


else if score > 50 THEN
    
    OUTPUT"You got a D-grade"


else if score > 40 THEN
    
    OUTPUT"You got an E-grade"


else
    
    OUTPUT"You got a U-grade"
```
